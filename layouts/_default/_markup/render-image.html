{{- $defaults := site.Params.site_images.defaults -}} {{- $addClass := $defaults.imageAddClass | default "img-fluid lazyload blur-up" -}} {{- $sizesAttr := $defaults.sizes | default "auto" -}} {{- /* Настраиваем счетчик и авто-alt на основе заголовка страницы */}} {{- $pageTitle := .Page.Title -}} {{- $imgIndex := .Page.Scratch.Get "image_counter" | default 0 -}} {{- $imgIndex = add $imgIndex 1 -}} {{- .Page.Scratch.Set "image_counter" $imgIndex -}} {{- $altText := .PlainText -}} {{- if not $altText -}} {{- $altText = printf "%s - %d" $pageTitle $imgIndex -}} {{- end -}} {{- $titleText := .Title | default $altText -}} {{- /* Проверяем, является ли ссылка внешней (начинается с http/https) */}} {{- $isRemote := or (hasPrefix .Destination "http://") (hasPrefix .Destination "https://") -}} {{- if $isRemote -}} {{- /* ВАРИАНТ 1: Внешнее изображение */}} {{- else -}} {{- /* ВАРИАНТ 2: Локальное изображение */}} {{- $rawSrc := .Destination -}} {{- /* Очищаем пути для безопасного поиска */}} {{- $cleanSrc := strings.TrimPrefix "/" $rawSrc -}} {{- $cleanSrc = strings.TrimPrefix "./" $cleanSrc -}} {{- /* Инициализируем переменную пустой, чтобы она была доступна везде ниже */}} {{- $image := "" -}} {{- /* 1. Пытаемся найти в ресурсах страницы (Page Bundle) */}} {{- $image = .Page.Resources.GetMatch $rawSrc -}} {{- /* 2. Если не нашли, ищем глобально в assets/ */}} {{- if not $image -}} {{- $assetsSrc := strings.TrimPrefix "assets/" $cleanSrc -}} {{- $image = resources.Get $assetsSrc -}} {{- end -}} {{- if $image -}} {{- /* Картинка успешно найдена в ресурсах, обрабатываем форматы */}} {{- $ext := lower (path.Ext $image.Name) -}} {{- if or (eq $ext ".svg") (eq $ext ".gif") }} {{- /* Для SVG и GIF ресайз не нужен */}} {{- else -}} {{- /* Для стандартных растровых картинок (PNG, JPG, WebP) */}} {{- /* УМНАЯ ОБРАБОТКА ДЕФОЛТНОГО ПРАВИЛА ИЗ КОНФИГА (940px) */}} {{- $processRule := $defaults.process | default "resize 940x" -}} {{- $processedImage := "" -}} {{- if hasPrefix $processRule "resize" -}} {{- $cleanRule := strings.TrimPrefix "resize " $processRule -}} {{- $processedImage = $image.Resize $cleanRule -}} {{- else if hasPrefix $processRule "fill" -}} {{- $cleanRule := strings.TrimPrefix "fill " $processRule -}} {{- $processedImage = $image.Fill $cleanRule -}} {{- else -}} {{- $processedImage = $image.Resize $processRule -}} {{- end -}} {{- /* ГЕНЕРАЦИЯ АДАПТИВНЫХ ОПТИМИЗИРОВАННЫХ НАБОРОВ SHIFT-WIDTHS (webp q75) */}} {{- $imgSrcSet := slice -}} {{- $widths := $defaults.widths | default (slice 480 768 940 1024 1366 1600) -}} {{- range $widths -}} {{- if le . $image.Width -}} {{- $webpUrl := (printf "%dx webp q75" . | $image.Resize).RelPermalink -}} {{- $imgSrcSet = $imgSrcSet | append (printf "%s %dw" $webpUrl .) -}} {{- end -}} {{- end -}} {{- $imgSrcSet = (delimit $imgSrcSet ", ") -}} {{- /* Финальный src (дефолтный размер 940px, сжатый в webp q80) */}} {{- $finalSrc := (printf "%dx%d webp q80" $processedImage.Width $processedImage.Height | $processedImage.Resize).RelPermalink -}} {{- end -}} {{- else -}} {{- /* ФОЛБЕК: Если файл лежит в папке static/ или просто не найден в ресурсах */}} {{- end -}} {{- end -}}
